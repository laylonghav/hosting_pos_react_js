import { Button } from "@/components/ui/button";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from "@/components/ui/dialog";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import { Spinner } from "@/components/ui/spinner";
import {
  Table,
  TableBody,
  TableCell,
  TableHead,
  TableHeader,
  TableRow,
} from "@/components/ui/table";
import { formatDate } from "@/utils/helpers/formats";
import { request } from "@/utils/request/request";
import { Edit, Plus, Search, SearchSlash, Trash } from "lucide-react";
import { useEffect, useState } from "react";

export default function CategoryPage() {
  const [category, setCategory] = useState([]);
  const [isOpen, setIsOpen] = useState(false);
  const [loading, setLoading] = useState(false);
  const [isEdit, setIsEdit] = useState(false);
  const [sellectedItem, setsellectedItem] = useState(null);
  const [isOpenDelete, setIsOpenDelete] = useState(false);
  const [query, setQuery] = useState("");
  const [form, setForm] = useState({
    id: "",
    name: "",
    description: "",
    status: true,
  });
  const onSubmit = async (e) => {
    e.preventDefault();

    try {
      if (isEdit) {
        const res = request(`category/${form.id}`, "put", form);
        if (res) {
          console.log(res);
          fectchingData();
        }
        setIsEdit(false);
      } else {
        const res = await request("category", "post", form);
        if (res) {
          console.log(res);
          fectchingData();
        }
      }
    } catch (error) {
      console.log(error);
    }

    console.log(form);
    setIsOpen(false);
    setForm({
      name: "",
      description: "",
      status: true,
    });
  };
  const fectchingData = async () => {
    setLoading(true);
    const res = await request("category", "get");
    if (res) {
      setLoading(false);
      console.log(res);
      setCategory(res.data);
    }
  };

  useEffect(() => {
    fectchingData();
  }, []);

  const onDelte = (itemDelete) => {
    console.log(itemDelete);
    setsellectedItem(itemDelete);
    setIsOpenDelete(true);
  };
  const onEdit = (itemEdit) => {
    console.log(itemEdit);
    setForm(itemEdit);
    setIsOpen(true);
    setIsEdit(true);
  };

  const confirmDeleted = () => {
    try {
      const res = request(`category/${sellectedItem?.id}`, "delete");
      if (res) {
        console.log(res);
        fectchingData();
        setIsOpenDelete(false);
      }
    } catch (error) {
      console.log(error);
    }
  };

  const table_head = ["Id", "Name", "Description", "Status", "Actions"];
  return (
    <div>
      <div className="flex justify-between">
        <div className="flex gap-3 items-center">
          <h1>Category</h1>
          <Input
            placeholder="Search Category"
            value={query}
            onChange={(e) => setQuery(e.target.value)}
          />
          <Button
            onClick={async () => {
              setLoading(true);
              const res = await request(`category/search/?q=${query}`, "get");
              if (res) {
                setLoading(false);
                console.log(res);
                setCategory(res.data);
              }
            }}
          >
            <Search />
          </Button>
          <Button
            onClick={() => {
              fectchingData();
              setQuery("");
            }}
          >
            <SearchSlash />
          </Button>
        </div>

        <Dialog open={isOpen} onOpenChange={setIsOpen}>
          <DialogTrigger>
            <Button>
              <Plus />
            </Button>
          </DialogTrigger>
          <DialogContent>
            <DialogHeader>
              <DialogTitle>Create Category</DialogTitle>
            </DialogHeader>

            <form action="" onSubmit={onSubmit}>
              <div className="flex flex-col gap-5">
                <div className="flex flex-col gap-3">
                  <Label>Name</Label>
                  <Input
                    value={form.name}
                    onChange={(e) => setForm({ ...form, name: e.target.value })}
                    placeholder="Name"
                    required
                  />
                </div>
                <div className="flex flex-col gap-3">
                  <Label>Description</Label>
                  <Input
                    value={form.description}
                    onChange={(e) =>
                      setForm({ ...form, description: e.target.value })
                    }
                    placeholder="Description"
                    required
                  />
                </div>
                <div className="flex flex-col gap-3">
                  <Label>Status</Label>
                  <Select
                    value={String(form.status)}
                    onValueChange={(value) =>
                      setForm({ ...form, status: value === "true" })
                    }
                  >
                    <SelectTrigger className="w-full">
                      <SelectValue placeholder="Pls select status" />
                    </SelectTrigger>
                    <SelectContent>
                      <SelectItem value="true">Active</SelectItem>
                      <SelectItem value="false">Inactive</SelectItem>
                    </SelectContent>
                  </Select>
                </div>
              </div>
              <div className="mt-4 flex justify-end">
                <div className="flex gap-3">
                  <Button
                    type="button"
                    onClick={() => {
                      setForm({
                        id: "",
                        name: "",
                        description: "",
                        status: true,
                      });
                      setIsEdit(false);
                      setIsOpen(false);
                    }}
                  >
                    Close
                  </Button>
                  <Button type="submit">Save</Button>
                </div>
              </div>
            </form>
          </DialogContent>
        </Dialog>
      </div>

      <Dialog open={isOpenDelete} onOpenChange={setIsOpenDelete}>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>
              Ara you sure you want to delete this {sellectedItem?.name} ?
            </DialogTitle>
          </DialogHeader>
          <div className="flex justify-end gap-3 mt-5">
            <Button
              onClick={() => {
                setIsOpenDelete(false);
                setsellectedItem(null);
              }}
              variant={"outline"}
            >
              Cancel
            </Button>
            <Button onClick={confirmDeleted} variant={"destructive"}>
              Delete
            </Button>
          </div>
        </DialogContent>
      </Dialog>
      <div className="mt-5">
        <Table>
          <TableHeader>
            <TableRow>
              {table_head.map((item, index) => (
                <TableHead key={index}>{item}</TableHead>
              ))}
            </TableRow>
          </TableHeader>
          <TableBody>
            {loading ? (
              <TableRow>
                <TableCell colSpan={5}>
                  <div className={"flex justify-center mt-6"}>
                    <Spinner className={"size-8"} />
                  </div>
                </TableCell>
              </TableRow>
            ) : (
              <>
                {category?.map((item, index) => (
                  <TableRow>
                    <TableCell key={index}>{index + 1}</TableCell>
                    <TableCell>{item?.name}</TableCell>
                    <TableCell>{item?.description}</TableCell>

                    <TableCell>
                      {item?.status ? "Active" : "Inactive"}
                    </TableCell>
                    <TableCell>{formatDate(item?.created_at)}</TableCell>
                    <TableCell>{formatDate(item?.updated_at)}</TableCell>
                    <TableCell>
                      <div className="flex gap-5">
                        <Button onClick={() => onEdit(item)}>
                          <Edit />
                        </Button>
                        <Button onClick={() => onDelte(item)}>
                          <Trash />
                        </Button>
                      </div>
                    </TableCell>
                  </TableRow>
                ))}
              </>
            )}
          </TableBody>
        </Table>
      </div>
    </div>
  );
}
